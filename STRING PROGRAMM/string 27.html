     void displayAlignedText(char text[], int alignment) {
          int length = strlen(text), i;
      
          if (alignment == 1) {
              printf("%s\n", text); 
          } else if (alignment == 2) {
              int spaces = (80 - length) / 2; 
          for (i = 0; i < spaces; i++) {
              printf(" ");
          }
          printf("%s\n", text);
          } else if (alignment == 3) {
          int spaces = 80 - length;
          for (i = 0; i < spaces; i++) {
              printf(" ");
          }
          printf("%s\n", text);
          } else {
          printf("Invalid alignment option.\n");
          }
      }
      
      void main() {
          char text[25];
          int alignment;
          clrscr();
          printf("Enter String\n");
          gets(text);
          printf("Enter alignment option (1 for left, 2 for center, 3 for right): ");
          scanf("%d", &alignment);
      
          displayAlignedText(text, alignment);
